#include <iostream>
using namespace std;
    
    //function
    bool isSymmetric(int arr[],int n){
        //set sym to true
        bool sym=true;
        //loop to run through entire array elements
        for(int i=0; i<n;i++){
            //check first and last element,then second and second last and so on...
            if(arr[i]!=arr[n-i-1]){
                //if both the value are not equal set sym to false
                sym=false;
                //and break the loop because there is no need to check further numbers
                return sym;
            }
        }
        //if all of them are equal then return sym as true
        return sym;
    }

int main()
{
    //assign n with length of array
    int n=9;
    //decalare arr of size n
    int arr[n];
    //prompt user from array elements
    cout<<"Enter 9 elements"<<endl;
    for (int i=0;i<n;i++){
        cin>>arr[i];
    }
    //isSymmetric will return 0 if Array is not Symmetric else 1 if Symmetric
    if(isSymmetric(arr,n)==0){
        cout<<"Array is not Symmetric";
    }
    else{
        cout<<"Array is Symmetric";
    }
    return 0;
}
